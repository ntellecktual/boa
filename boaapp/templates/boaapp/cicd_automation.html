# NeonDeck Entertainment — Databricks Platform Deployment Pipeline
# Organization: neondeck-gaming  |  Project: DataPlatform  |  AZ-400 compliant

trigger:
  branches:
    include: [main, release/*]
  paths:
    include: [databricks/**, terraform/**, pipelines/**]

variables:
  - group: neondeck-databricks-prod    # Key Vault-linked variable group
  - group: neondeck-entra-svc-conn     # Entra service principal credentials
  - name: databricksWorkspace
    value: adb-neondeck-prod-eastus2

stages:
  - stage: Validate
    displayName: Validate & Lint
    jobs:
      - job: ValidateNotebooks
        pool: { vmImage: ubuntu-latest }
        steps:
          - task: UsePythonVersion@0
            inputs: { versionSpec: '3.11' }
          - script: pip install databricks-cli dbx ruff
          - script: ruff check notebooks/ pipelines/
            displayName: Lint Python notebooks
          - script: databricks fs ls dbfs:/neondeck
            displayName: Validate DBFS connectivity
            env:
              DATABRICKS_HOST: $(databricksHost)
              DATABRICKS_TOKEN: $(databricksToken)  # resolved from Key Vault VG

  - stage: Terraform
    displayName: Infrastructure Plan + Apply
    dependsOn: Validate
    jobs:
      - deployment: TerraformApply
        environment: production   # requires approval gate
        strategy:
          runOnce:
            deploy:
              steps:
                - task: TerraformTaskV4@4
                  inputs:
                    command: plan
                    workingDirectory: $(System.DefaultWorkingDirectory)/terraform
                    backendServiceArm: neondeck-azure-svc-conn
                    backendAzureRmResourceGroupName: rg-neondeck-tfstate
                    backendAzureRmStorageAccountName: ndtfstateprod
                - task: TerraformTaskV4@4
                  inputs: { command: apply, commandOptions: -auto-approve }

  - stage: DatabricksDeploy
    displayName: Deploy to Databricks
    dependsOn: Terraform
    jobs:
      - job: DeployNotebooks
        steps:
          - script: dbx deploy --environment=production --cluster-id=$(clusterIdProd)
          - script: databricks jobs reset --job-id=$(playerEtlJobId) --json @jobs/player_etl.json
            displayName: Update Player ETL DLT job definition
          - task: AzureCLI@2
            displayName: Sync Entra groups → Unity Catalog grants
            inputs:
              azureSubscription: neondeck-azure-svc-conn
              scriptType: pscore
              scriptPath: scripts/Sync-EntraGroupsToUnityCatalog.ps1
# terraform/databricks.tf — NeonDeck Databricks Workspace & Cluster Policies

resource "azurerm_databricks_workspace" "neondeck_prod" {
  name                = "adb-neondeck-prod-eastus2"
  resource_group_name = azurerm_resource_group.platform.name
  location            = "East US 2"
  sku                 = "premium"
  custom_parameters {
    virtual_network_id  = azurerm_virtual_network.platform.id
    public_subnet_name  = "snet-databricks-public"
    private_subnet_name = "snet-databricks-private"
    no_public_ip        = true  # Secure Cluster Connectivity mode
  }
}

resource "databricks_cluster_policy" "gaming_analytics" {
  name = "NeonDeck-GamingAnalytics-Policy"
  definition = jsonencode({
    "spark_version"            : { "type": "fixed",     "value": "14.3.x-scala2.12" },
    "node_type_id"             : { "type": "allowlist", "values": ["Standard_DS4_v2", "Standard_DS5_v2"] },
    "autoscale.min_workers"    : { "type": "range",     "minValue": 2, "maxValue": 4 },
    "autoscale.max_workers"    : { "type": "range",     "minValue": 4, "maxValue": 20 },
    "azure_attributes.availability": { "type": "fixed", "value": "SPOT_WITH_FALLBACK_AZURE" },
    "spark_conf.spark.databricks.io.cache.enabled": { "type": "fixed", "value": "true" }
  })
}

resource "databricks_metastore_assignment" "prod" {
  metastore_id = databricks_metastore.neondeck.id
  workspace_id = azurerm_databricks_workspace.neondeck_prod.workspace_id
}

resource "databricks_grants" "catalog_gaming" {
  catalog = databricks_catalog.gaming.name
  grant { principal = "data-engineers@neondeck.com"; privileges = ["USE_CATALOG", "CREATE_SCHEMA"] }
  grant { principal = "analysts@neondeck.com";      privileges = ["USE_CATALOG", "SELECT"] }
  grant { principal = "compliance@neondeck.com";   privileges = ["USE_CATALOG", "SELECT"] }
}

# PowerShell: sync Key Vault secrets into ADO Variable Group (nightly)
Sync-AzKeyVaultToDevOpsVariableGroup.ps1:
  $secrets = Get-AzKeyVaultSecret -VaultName "kv-neondeck-prod"
  foreach ($s in $secrets) {
    Update-AzDevOpsVariableGroup -Org "neondeck-gaming" `
      -GroupName "neondeck-databricks-prod" `
      -Variable $s.Name -IsSecret $true
  }

Entra ID

Identity & Access Management

Azure Entra ID governs all access to Azure DevOps, Databricks, and data assets — with conditional access policies enforcing MFA, device compliance, and just-in-time PIM roles.

Enterprise SSO
SAML 2.0 & OIDC for Azure DevOps, Databricks, and 18 SaaS apps. App registrations with client credential flow for unattended service principals.
Conditional Access
Require MFA + compliant device for all Databricks access. Geo-block from non-licensed gaming jurisdictions. Session risk triggers step-up auth.
PIM Just-in-Time
Privileged Identity Management for Owner/Contributor roles. Max 4-hour activation, approval required, all activations logged to Azure Monitor.
SCIM Provisioning
Automated user & group lifecycle: Entra group changes propagate to Databricks Unity Catalog permissions and Azure DevOps project access within 5 min.
Access Reviews
Quarterly automated Entra access reviews for all privileged roles. Non-responded reviews auto-revoke. Integrated with SOC 2 audit evidence collection.
Managed Identities
System-assigned MI on ADF, AKS, and Functions eliminates stored creds. Databricks service principals use Entra Workload Identity Federation.
